192 void InterDiffusionCoefficient::get_property(
193  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<mfem::Vector>>>&
194  output_system,
195  std::vector<std::tuple<std::vector<std::string>, mfem::Vector>> input_system) {
196  primary_mob_gf_.clear();
197  secondary_mob_gf_.clear();
198 
199  x_gf_.clear();
200  phi_gf_.clear();
201  for (const auto& [aux_infos, aux_gf] : input_system) {
202  if (!aux_infos.empty()) {
203  const int size = aux_infos.size();
204  const std::string_view type_info_view(aux_infos.back());
205 
206  if (!type_info_view.empty() && type_info_view.starts_with("mob")) {
207  const std::string phase_info(aux_infos[size - 3]);
208  const std::string elem_info(aux_infos[size - 2]);
209  if (phase_info == this->primary_phase_) {
210  this->primary_mob_gf_.emplace(toUpperCase(elem_info), aux_gf);
211 
212  } else if (phase_info == this->secondary_phase_) {
213  this->secondary_mob_gf_.emplace(toUpperCase(elem_info), aux_gf);
214  }
215  }
216  if (!type_info_view.empty() && type_info_view == "x") {
217  const std::string elem_info(aux_infos[size - 2]);
218  this->x_gf_.emplace(toUpperCase(elem_info), aux_gf);
219  }
220  if (!type_info_view.empty() && type_info_view == "phi") {
221  const std::string elem_info(aux_infos[size - 2]);
222  this->phi_gf_.emplace_back(aux_gf);
223  }
224  }
225  }
226  const int size_gf = this->primary_mob_gf_.at(this->first_component_).Size();
227 
228  auto mult_vectors = [](const std::vector<double>& v1, const std::vector<double>& v2,
229  std::vector<double>& out) {
230  std::ranges::transform(v1, v2, out.begin(), [](double x, double y) { return x * y; });
231  };
232 
233  auto sum_vectors = [](const std::vector<double>& v1, const std::vector<double>& v2,
234  std::vector<double>& out) {
235  std::ranges::transform(v1, v2, out.begin(), [](double x, double y) { return x + y; });
236  };
237 
238  auto scale_vector = [](std::vector<double>& v, double scalar) {
239  std::ranges::transform(v, v.begin(), [=](double x) { return x * scalar; });
240  };
241 
242  auto add_scalar = [](std::vector<double>& v, double scalar) {
243  std::ranges::transform(v, v.begin(), [=](double x) { return x + scalar; });
244  };
245 
246  std::vector<std::vector<double>> vmob;
247  vmob.reserve(this->primary_mob_gf_.size());
248 
249  //
250  std::vector<double> phi(size_gf, 1.0);
251  if (!this->single_phase_) {
252  for (int k = 0; k < size_gf; k++) {
253  phi[k] = std::clamp(this->phi_gf_[0](k), 0.0, 1.0);
254  }
255  }
256  std::vector<double> xo(this->x_gf_.at(this->first_component_).Size(), 0.);
257  for (int k = 0; k < this->x_gf_.at(this->first_component_).Size(); k++) {
258  xo[k] = this->x_gf_.at(this->first_component_)(k);
259  }
260 
261  std::vector<double> mobo(this->primary_mob_gf_.at(this->first_component_).Size(), 0.);
262  if (this->single_phase_) {
263  for (int k = 0; k < this->primary_mob_gf_.at(this->first_component_).Size(); k++) {
264  mobo[k] = this->primary_mob_gf_.at(this->first_component_)(k);
265  }
266  } else {
267  for (int k = 0; k < this->primary_mob_gf_.at(this->first_component_).Size(); k++) {
268  mobo[k] = phi[k] * this->primary_mob_gf_.at(this->first_component_)(k);
269 
270  mobo[k] += (1. - phi[k]) * this->secondary_mob_gf_.at(this->first_component_)(k);
271  }
272  }
273 
274  std::vector<double> mob(size_gf, 0.0);
275  std::vector<double> sum_mob(size_gf, 0.0);
276 
277  for (const auto& [compo, mobi] : this->primary_mob_gf_) {
278  if (compo == this->first_component_) continue;
279 
280  std::vector<double> xj(size_gf, 0.0);
281  std::vector<double> mobj(size_gf, 0.0);
282 
283  if (compo == this->last_component_) {
284  add_scalar(xj, -1.0);
285  for (const auto& [other_compo, _] : this->primary_mob_gf_) {
286  if (other_compo != this->last_component_) {
287  for (int k = 0; k < this->x_gf_.at(other_compo).Size(); k++) {
288  xj[k] += this->x_gf_.at(other_compo)(k);
289  }
290  }
291  }
292  scale_vector(xj, -1.0);
293  } else {
294  for (int k = 0; k < this->x_gf_.at(compo).Size(); k++) {
295  xj[k] += this->x_gf_.at(compo)(k);
296  }
297  }
298 
299  mult_vectors(mobo, xj, mob);
300  if (this->single_phase_) {
301  for (int k = 0; k < mobi.Size(); k++) {
302  mobj[k] = mobi(k) * xo[k];
303  }
304  } else {
305  for (int k = 0; k < mobi.Size(); k++) {
306  mobj[k] = (mobi(k) * phi[k] + (1. - phi[k]) * this->secondary_mob_gf_.at(compo)(k)) * xo[k];
307  }
308  }
309  sum_vectors(mob, mobj, mob);
310  mult_vectors(mob, xj, mob);
311  mult_vectors(mob, xo, mob);
312  sum_vectors(sum_mob, mob, sum_mob);
313  scale_vector(mob, -1.0);
314  vmob.emplace_back(mob);
315  }
316  vmob.insert(vmob.begin(), sum_mob);
318  int j = 0;
319  for (auto& [output_infos, output_value] : output_system) {
320  mfem::Vector vv(vmob[j].size());
321  for (int k = 0; k < vmob[j].size(); k++) {
322  vv(k) = vmob[j][k];
323  std::string msg = "Error for j " + std::to_string(j) + " at node " + std::to_string(k) +
324  " with mob " + std::to_string(vv(k)) + " phi " + std::to_string(phi[k]);
325  MFEM_VERIFY(vv(k) < 1., msg.c_str());
326  }

328  j++;
