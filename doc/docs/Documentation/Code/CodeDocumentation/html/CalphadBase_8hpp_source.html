203 template <typename T>
204 void CalphadBase<T>::global_execute(
205  const int dt, const double time_step, const std::vector<T>& tp_gf,
206  const std::vector<std::tuple<std::string, std::string>>& chemicalsystem,
207  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<T>>>& output_system,
208  const std::vector<std::tuple<std::vector<std::string>, mfem::Vector>>& previous_output_system,
209  std::optional<const std::tuple<std::string, T, T>> phase_field_gf,
210  std::optional<const std::vector<T>> tp_gf_old,
211  std::optional<const std::vector<std::tuple<std::string, std::string, T, T>>> x_gf,
212  std::optional<const std::vector<std::tuple<std::string, T>>> coordinates) {
213  const size_t nb_nodes = this->CU_->get_size(tp_gf[0]);
214  // Reinitialize containers
215  this->clear_containers();
216 
217  // Execute
218  if (!this->is_KKS_) {
219  // Creation list of nodes
220  std::set<int> list_nodes;
221  for (int i = 0; i < nb_nodes; ++i) {
222  list_nodes.insert(i);
223  }
224  this->execute(dt, list_nodes, tp_gf, chemicalsystem);
225  } else {
226  // Verify that all required fields are available for KKS execution
227  MFEM_VERIFY(phase_field_gf.has_value(),
228  "Error: phase_fields_gf is required for KKS execution.");
229  MFEM_VERIFY(tp_gf_old.has_value(), "Error: tp_gf_old is required for KKS execution.");
230  MFEM_VERIFY(x_gf.has_value(), "Error: x_gf is required for KKS execution.");
231  MFEM_VERIFY(coordinates.has_value(), "Error: coordinates is required for KKS execution.");
232 
233  // Execute KKS linearization
234  this->KKS_->execute_linearization(*this, dt, time_step, tp_gf, *tp_gf_old, *phase_field_gf,
235  chemicalsystem, *x_gf, *coordinates);
236  }
237  // Use specific CALPHAD C++ containers to update output_system
238  this->update_outputs(dt, nb_nodes, output_system, previous_output_system);
239 }
