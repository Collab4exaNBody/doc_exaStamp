164  // elvect = 0.0;

166  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
167  el[blk]->CalcShape(ip, Psi); //

171 
172  el[blk]->CalcPhysDShape(Tr, gradPsi);
173  gradPsi.MultTranspose(*elfun[blk], gradU);
