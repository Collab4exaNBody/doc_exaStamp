132 template <class VARS>
133 void DiffusionFluxNLFormIntegrator<VARS>::AssembleElementVector(
134  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
135  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
136  int blk = 0;
137  int nd = el[blk]->GetDof();
138  int dim = el[blk]->GetDim();
139  gradPsi.SetSize(nd, dim);
140  Psi.SetSize(nd);
141  int nElement = Tr.ElementNo;
142 
143  this->Flux_.SetSize(dim);
144 
145  elvect[blk]->SetSize(nd);
146  *elvect[blk] = 0.;
147 
148  mfem::Vector grad_uold;
149  grad_uold.SetSize(dim);
150 
151  const mfem::IntegrationRule* ir =
152  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
153 
154  // Loop over integration points
155  for (int i = 0; i < ir->GetNPoints(); i++) {
156  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
157 
158  el[blk]->CalcShape(ip, Psi);
159  Tr.SetIntPoint(&ip);
160 
161  const auto& u = *elfun[blk] * Psi;
162  // Stabilization contribution : D_stab * (Grad u - Grad un)
163  el[blk]->CalcPhysDShape(Tr, this->gradPsi);
164  this->gradPsi.MultTranspose(*elfun[blk], this->Flux_);
165  this->u_old_[blk].GetGradient(Tr, this->gradPsi, grad_uold);
166 
167  this->Flux_.Add(-1, grad_uold);
168  this->Flux_ *= this->coeff_stab_;
169 
170  // Diffusion flux (see child classes)
171  this->add_diffusion_flux(Tr, nElement, ip, dim);
172 
173  this->Flux_ *= ip.weight * Tr.Weight();
174 
175  this->gradPsi.AddMult(this->Flux_, *elvect[blk], 1.0);
176  }
177 }
