154 template <class T, int DIM, class NLFI, class LHS_NLFI,
155  template <class, int, class, class> class OPEBASE>
156 void PhaseFieldOperatorBase<T, DIM, NLFI, LHS_NLFI, OPEBASE>::ComputeEnergies(
157  const int& it, const double& t, const double& dt, const std::vector<mfem::Vector>& u) {
158  Catch_Time_Section("PhaseFieldOperatorBase::ComputeEnergies");
159 
160  std::vector<mfem::ParGridFunction> vun;
161  vun.reserve(u.size());
162  for (size_t i = 0; i < u.size(); ++i) {
163  mfem::ParGridFunction un(this->fes_[i]);
164  un.SetFromTrueDofs(u[i]);
165  vun.emplace_back(std::move(un));
166  }
167 
168  std::vector<mfem::ParGridFunction*> vun_ptr;
169  for (auto& un : vun) {
170  vun_ptr.push_back(&un);
171  }
172 
173  auto gphi = this->nlfi_ptr_->get_energy(vun_ptr, this->omega_);
174 
175  std::unique_ptr<GradientEnergyCoefficient> gradGphi =
176  this->nlfi_ptr_->get_grad_energy(vun_ptr, 0.5 * this->lambda_);
177  auto energy_coeff = gphi.get();
178 
179  mfem::ParGridFunction gf(this->fes_[0]);
180  gf.ProjectCoefficient(*energy_coeff);
181 
182  mfem::ParGridFunction sigf(this->fes_[0]);
183  auto grad_energy_coeff = gradGphi.get();
184  sigf.ProjectCoefficient(*grad_energy_coeff);
185 
186  mfem::ConstantCoefficient zero(0.);
187  const auto energy = gf.ComputeLpError(1, zero);
188  const auto interfacial_energy = sigf.ComputeLpError(1, zero);
189 
190  this->time_specialized_.emplace(IterationKey(it, dt, t),
191  SpecializedValue("Density[J.m-3]", energy + interfacial_energy));
192  this->time_specialized_.emplace(IterationKey(it, dt, t),
193  SpecializedValue("Sigma[J.m-3]", 2. * interfacial_energy));
194 }
