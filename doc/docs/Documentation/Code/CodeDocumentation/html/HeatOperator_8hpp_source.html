153  const int& it, const double& t, const double& dt, const std::vector<mfem::Vector>& u) {
154  Catch_Time_Section("HeatOperatorBase::ComputeEnergies");
155 
156  std::vector<mfem::ParGridFunction> vun;
157  vun.reserve(u.size());
158  for (size_t i = 0; i < u.size(); ++i) {
159  mfem::ParGridFunction un(this->fes_[i]);
160  un.SetFromTrueDofs(u[i]);
161  vun.emplace_back(std::move(un));
162  }
163 
164  std::vector<mfem::ParGridFunction*> vun_ptr;
165  for (auto& un : vun) {
166  vun_ptr.push_back(&un);
167  }
168 
169  auto gphi = this->nlfi_ptr_->get_energy(vun_ptr, 1.);
170  auto energy_coeff = gphi.get();
171  mfem::ParGridFunction gf(this->fes_[0]);
172 
173  gf.ProjectCoefficient(*energy_coeff);
174 
175  // Calcul de l'int√©grale de l'objet FunctionCoefficient sur le domaine
176  mfem::ConstantCoefficient zero(0.);
177  const auto energy = gf.ComputeLpError(1, zero);
178  this->time_specialized_.emplace(IterationKey(it, dt, t),
179  SpecializedValue("Density[J.m-3]", energy));
