225 FType BlockAllenCahnNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::double_well_derivative(
226  const int order_derivative, mfem::ElementTransformation& Tr, const mfem::IntegrationPoint& ir) {
227  return FType([this, order_derivative, &Tr, &ir](const double& u) {
228  const auto& un = this->u_old_[0].GetValue(Tr, ir);
229  FType W_derivative;
230  if (order_derivative == 1) {
231  W_derivative = this->energy_first_derivative_potential_.getPotentialFunction(un);
232  } else if (order_derivative == 2) {
233  W_derivative = this->energy_second_derivative_potential_.getPotentialFunction(un);
234  } else {
235  std::runtime_error("Error while setting the order of derivative : only 1 and 2 are allowed.");
236  }
237  const double omega = this->omega(Tr, ir, u, this->params_);
238 
239  const auto& w_prime = omega * W_derivative(u);
240  return w_prime;
241  });
242 }

323  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
324  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
325  //
326  // Block 0 R(phi)=eta psi + w'psi + lambda grad phi grad psi
327  //
328  {
329  int blk = 0;
330  int off_blk = 1;
331  mfem::DenseMatrix gradPsi;
332  mfem::Vector Psi, gradU;
333  int nd = el[blk]->GetDof();
334  int dim = el[blk]->GetDim();
335  gradPsi.SetSize(nd, dim);
336  Psi.SetSize(nd);
337  gradU.SetSize(dim);
338  elvect[blk]->SetSize(nd);
339  *elvect[blk] = 0.;
340  const mfem::IntegrationRule* ir =
341  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
343  for (int i = 0; i < ir->GetNPoints(); i++) {
344  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
345  el[blk]->CalcShape(ip, Psi); //
346  Tr.SetIntPoint(&ip);
347 
348  const auto& eta = *elfun[off_blk] * Psi;
349 
350  const auto& phi = *elfun[blk] * Psi;
351 
352  const double xx = ip.weight * Tr.Weight();
353  el[blk]->CalcPhysDShape(Tr, gradPsi);
354  gradPsi.MultTranspose(*elfun[blk], gradU);
355  const double lambda = this->lambda(Tr, ip, phi, this->params_);
356 
357  gradU *= xx * lambda;
358  gradPsi.AddMult(gradU, *elvect[blk]);
359 
360  const double ww = xx * (eta + this->energy_derivatives(1, Tr, ip)(phi));
361 
362  add(*elvect[blk], ww, Psi, *elvect[blk]);
363  }
364  }
365  //
366  // Block 1 R(eta) = -M * eta
367  //
368  {
369  int blk = 1;
370  int off_blk = 0;
371  mfem::DenseMatrix gradPsi;
372  mfem::Vector Psi, gradU;
373  int nd = el[blk]->GetDof();
374  int dim = el[blk]->GetDim();
375  gradPsi.SetSize(nd, dim);
376  Psi.SetSize(nd);
377  gradU.SetSize(dim);
378  elvect[blk]->SetSize(nd);
379  *elvect[blk] = 0.;
380  const mfem::IntegrationRule* ir =
381  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
382 
383  for (int i = 0; i < ir->GetNPoints(); i++) {
384  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
385  el[blk]->CalcShape(ip, Psi); //
386  Tr.SetIntPoint(&ip);
387 
388  const auto& eta = *elfun[blk] * Psi;
389  const auto& phi = *elfun[off_blk] * Psi;
390 
391  const double coef_mob = this->mobility(Tr, ip, phi, this->params_) * ip.weight * Tr.Weight();
392 
393  const double ww = -coef_mob * eta;
394 
395  add(*elvect[blk], ww, Psi, *elvect[blk]);
396  }
397  }
