519 template <class T, int DIM, class NLFI, class LHS_NLFI>
520 void OperatorBase<T, DIM, NLFI, LHS_NLFI>::ComputeIntegral(
521  const int& it, const double& t, const double& dt, const int id_var, const std::string& name,
522  const mfem::Vector& u, const double lower_bound, const double upper_bound) {
523  Catch_Time_Section("OperatorBase::ComputeIntegral");
524 
525  mfem::ParGridFunction gf(this->fes_[id_var]);
526  mfem::Vector u_cut(u.Size());
527  std::transform(u.begin(), u.end(), u_cut.begin(), [&](auto value) {
528  if (value > upper_bound || value < lower_bound) {
529  return 0.0;
530  }
531  return value;
532  });
533  gf.SetFromTrueDofs(u_cut);
534 
535  double integral = 0.0;
536  double domain_volume = 0.0;
537  mfem::Vector vals;
538  const mfem::FiniteElement* fe;
539  mfem::ElementTransformation* Tr;
540 
541  for (int i = 0; i < this->fes_[id_var]->GetNE(); i++) {
542  fe = this->fes_[id_var]->GetFE(i);
543  const mfem::IntegrationRule* ir;
544 
545  int intorder = 2 * fe->GetOrder() + 3; // <----------
546  ir = &(mfem::IntRules.Get(fe->GetGeomType(), intorder));
547 
548  mfem::real_t int_elem = 0.0;
549  mfem::real_t domain_volume_elem = 0.0;
550 
551  gf.GetValues(i, *ir, vals);
552  Tr = this->fes_[id_var]->GetElementTransformation(i);
553  for (int j = 0; j < ir->GetNPoints(); j++) {
554  const mfem::IntegrationPoint& ip = ir->IntPoint(j);
555  Tr->SetIntPoint(&ip);
556 
557  int_elem += vals(j) * ip.weight * Tr->Weight();
558  domain_volume_elem += ip.weight * Tr->Weight();
559  }
560 
561  integral += int_elem;
562  domain_volume += domain_volume_elem;
563  }
564 
565  mfem::real_t global_integral = 0.0;
566  mfem::real_t global_domain_volume = 0.0;
567  MPI_Allreduce(&integral, &global_integral, 1, mfem::MPITypeMap<mfem::real_t>::mpi_type, MPI_SUM,
568  MPI_COMM_WORLD);
569  MPI_Allreduce(&domain_volume, &global_domain_volume, 1, mfem::MPITypeMap<mfem::real_t>::mpi_type,
570  MPI_SUM, MPI_COMM_WORLD);
571 
572  integral = global_integral;
573  domain_volume = global_domain_volume;
574 
575  const double average = integral / domain_volume;
576  this->time_specialized_.emplace(IterationKey(it, dt, t),
577  SpecializedValue(name + "_integral[-]", integral));
578  this->time_specialized_.emplace(IterationKey(it, dt, t),
579  SpecializedValue(name + "_average[-]", average));
580 }

590 template <class T, int DIM, class NLFI, class LHS_NLFI>
591 void OperatorBase<T, DIM, NLFI, LHS_NLFI>::ComputeIsoVal(const int& it, const double& t,
592  const double& dt, const int var_id,
593  const std::string& var_name,
594  const mfem::Vector& u,
595  const double& iso_value) {
596  Catch_Time_Section("OperatorBase::ComputeIsoVal");
597  std::vector<std::string> vstr = {"x", "y", "z"};
598  std::multimap<IterationKey, SpecializedValue> variable_time_iso_specialized;
599 
600  if (!this->time_iso_specialized_[var_name].empty()) {
601  variable_time_iso_specialized.merge(this->time_iso_specialized_[var_name]);
602  }
603 
604  mfem::ParGridFunction gf(this->fes_[var_id]);
605  gf.SetFromTrueDofs(u);
606  std::vector<mfem::Vector> iso_points;
607 
608  for (int i = 0; i < this->fes_[var_id]->GetNE(); i++) {
609  const mfem::FiniteElement* el = this->fes_[var_id]->GetFE(i);
610  size_t dim = el->GetDim();
611  mfem::Array<int> dofs;
612  this->fes_[var_id]->GetElementDofs(i, dofs);
613 
614  mfem::DenseMatrix dof_coords;
615  mfem::ElementTransformation* Tr = this->fes_[var_id]->GetElementTransformation(i);
616  Tr->Transform(el->GetNodes(), dof_coords);
617  std::vector<double> dof_val;
618  dof_val.reserve(dofs.Size());
619  for (int j = 0; j < dofs.Size(); j++) {
620  dof_val.emplace_back(u(dofs[j]) - iso_value);
621  }
622 
623  // At least one positive and one negative value
624  bool has_positive_value =
625  std::any_of(dof_val.begin(), dof_val.end(), [](double x) { return x > 0; });
626  bool has_negative_value =
627  std::any_of(dof_val.begin(), dof_val.end(), [](double x) { return x < 0; });
628 
629  if (has_positive_value && has_negative_value) {
630  for (int j = 0; j < dofs.Size(); j++) {
631  mfem::Vector coord1(DIM);
632  double val1 = dof_val[j];
633  dof_coords.GetColumn(j, coord1);
634  for (int k = j + 1; k < dofs.Size(); k++) {
635  double val2 = dof_val[k];
636 
637  if (val1 * val2 < 0) {
638  const double abs_val1 = std::abs(val1);
639  const double abs_val2 = std::abs(val2);
640  double t = abs_val1 / (abs_val1 + abs_val2);
641 
642  mfem::Vector coord2(DIM), iso_coord(DIM);
643  iso_coord = mfem::infinity();
644  dof_coords.GetColumn(k, coord2);
645 
646  for (int d = 0; d < coord1.Size(); d++) {
647  iso_coord[d] = coord1[d] + t * (coord2[d] - coord1[d]);
648  }
649  iso_points.emplace_back(std::move(iso_coord));
650  }
651  }
652  }
653  for (size_t i = 0; i < iso_points.size(); i++) {
654  for (size_t d = 0; d < DIM; d++) {
655  variable_time_iso_specialized.emplace(IterationKey(it, dt, t),
656  SpecializedValue(vstr[d], iso_points[i](d)));
657  }
658  }
659  }
660  }
661  this->time_iso_specialized_[var_name] = variable_time_iso_specialized;
662 }
