228 FType AllenCahnGrainNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::double_well_derivative(
229  const int order_derivative, mfem::ElementTransformation& Tr, const mfem::IntegrationPoint& ir) {
230  return FType([this, order_derivative, &Tr, &ir](const double& u) {
231  const auto& un = this->u_old_[0].GetValue(Tr, ir);
232  FType W_derivative;
233  if (order_derivative == 1) {
234  W_derivative = this->energy_first_derivative_potential_.getPotentialFunction(un);
235  } else if (order_derivative == 2) {
236  W_derivative = this->energy_second_derivative_potential_.getPotentialFunction(un);
237  } else {
238  std::runtime_error("Error while setting the order of derivative : only 1 and 2 are allowed.");
239  }
240  const double omega = this->omega(Tr, ir, u, this->params_);
241 
242  const auto& w_prime = omega * W_derivative(u);
243  return w_prime;
244  });
245 }

277 void AllenCahnGrainNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::check_variables_consistency() {
278  this->aux_gf_ = this->get_aux_gf();
279  this->aux_old_gf_ = this->get_aux_old_gf();
280  this->aux_gf_infos_ = this->get_aux_infos();
281 
282  // Temperature scaling for mobility
283  bool temperature_found = false;
284  for (std::size_t i = 0; i < this->aux_gf_infos_.size(); ++i) {
285  const auto& variable_info = this->aux_gf_infos_[i];
286  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
287  size_t vsize = variable_info.size();
288 
289  MFEM_VERIFY(vsize >= 1,
290  "AllenCahnGrainNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>: at least "
291  "one additionnal information is expected for auxiliary variables associated with "
292  "this integrator");
293  const std::string& symbol = toUpperCase(variable_info.back());
294  if (symbol == "T") {
295  this->temp_gf_.emplace_back(std::move(this->aux_gf_[i]));
296  temperature_found = true;
297  break;
298  }
299  }
300  if (this->params_.has_parameter("ScaleMobilityByTemperature")) {
301  this->scale_mobility_by_temperature_ =
302  this->params_.template get_param_value<bool>("ScaleMobilityByTemperature");
303  if (this->scale_mobility_by_temperature_) {
304  MFEM_VERIFY(
305  temperature_found,
306  "AllenCahnGrainNLFormIntegrator: "
307  "Temperature variable required to scale mobility, but not found in auxiliary variables");
308  }
309  }

323 template <class VARS, ThermodynamicsPotentialDiscretization SCHEME, ThermodynamicsPotentials ENERGY,
324  Mobility MOBI>
325 void AllenCahnGrainNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::AssembleElementVector(
326  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
327  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
328  int num_blocks = el.Size();
329  for (int blk = 0; blk < num_blocks; ++blk) {
330  // Catch_Time_Section("AllenCahnGrainNLFormIntegrator:AssembleElementVector");
331  int nd = el[blk]->GetDof();
332  int dim = el[blk]->GetDim();
333  gradPsi.SetSize(nd, dim);
334  Psi.SetSize(nd);
335  gradU.SetSize(dim);
336  // elvect.SetSize(nd);
337  elvect[blk]->SetSize(nd);
338  *elvect[blk] = 0.;
339 
340  const mfem::IntegrationRule* ir =
341  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
342  // elvect = 0.0;
343  for (int i = 0; i < ir->GetNPoints(); i++) {
344  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
345  el[blk]->CalcShape(ip, Psi); //
346  Tr.SetIntPoint(&ip);
347 
348  const auto& u = *elfun[blk] * Psi;
349  // Interaction term : 2 * u_i (sum_j!=i u__j * u_j)
350  double interaction = 0.0;
351  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
352  interaction += *elfun[off_blk] * Psi;
353  }
354  interaction -= u * u;
355  interaction *= 2.0 * u;
356 
357  // Laplacian : given u, compute (grad(u), grad(psi)), psi is shape function.
358  // given u (elfun), compute grad(u)
359  el[blk]->CalcPhysDShape(Tr, gradPsi);
360  gradPsi.MultTranspose(*elfun[blk], gradU);
361  const double coef_mob = this->mobility(Tr, ip, u, this->params_) * ip.weight * Tr.Weight();
362  const double lambda = this->lambda(Tr, ip, u, this->params_);
363  gradU *= coef_mob * lambda;
364  gradPsi.AddMult(gradU, *elvect[blk]);
365 
366  // Given u, compute (w'(u) + interaction term , psi), psi is shape function
367  const double ww = coef_mob * (this->energy_derivatives(1, Tr, ip)(u) + interaction);
368  add(*elvect[blk], ww, Psi, *elvect[blk]);
369  }
370  }
371 }

384 template <class VARS, ThermodynamicsPotentialDiscretization SCHEME, ThermodynamicsPotentials ENERGY,
385  Mobility MOBI>
386 void AllenCahnGrainNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::AssembleElementGrad(
387  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
388  const mfem::Array<const mfem::Vector*>& elfun,
389  const mfem::Array2D<mfem::DenseMatrix*>& elmats) {
390  // Catch_Time_Section("AllenCahnGrainNLFormIntegrator::AssembleElementGrad");
391  int num_blocks = el.Size();
392 
393  // Diagonal
394  for (int blk = 0; blk < num_blocks; ++blk) {
395  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
396  // int nd = el.GetDof();
397  // int dim = el.GetDim();
398  int nd = el[blk]->GetDof();
399  int dim = el[blk]->GetDim();
400 
401  gradPsi.SetSize(nd, dim);
402  Psi.SetSize(nd);
403  // elmat.SetSize(nd);
404  elmats(blk, off_blk)->SetSize(nd);
405  *elmats(blk, off_blk) = 0.0;
406 
407  const mfem::IntegrationRule* ir =
408  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
409 
410  // elmat = 0.0;
411  for (int i = 0; i < ir->GetNPoints(); i++) {
412  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
413  el[blk]->CalcShape(ip, Psi);
414  const auto& u = *elfun[blk] * Psi;
415  const auto& v = *elfun[off_blk] * Psi;
416  const double coef_mob = this->mobility(Tr, ip, u, this->params_) * ip.weight * Tr.Weight();
417 
418  double interaction = 0.0;
419 
420  Tr.SetIntPoint(&ip);
421  double double_well = 0;
422  if (off_blk == blk) {
423  double_well = this->energy_derivatives(2, Tr, ip)(u);
424  el[blk]->CalcPhysDShape(Tr, gradPsi);
425  const double lambda = this->lambda(Tr, ip, u, this->params_);
426 
427  AddMult_a_AAt(coef_mob * lambda, gradPsi, *elmats(blk, off_blk));
428 
429  for (int off_blk = 0; off_blk < num_blocks; ++off_blk) {
430  interaction += *elfun[off_blk] * Psi;
431  }
432  interaction -= u * u;
433  interaction *= 2.0;
434  } else {
435  interaction = 4.0 * u * v;
436  }
437 
438  double fun_val = coef_mob * (double_well + interaction);
439  AddMult_a_VVt(fun_val, Psi, *elmats(blk, off_blk)); // w'(u)*(du, psi)
440  }
441  }
442  }
443 }
