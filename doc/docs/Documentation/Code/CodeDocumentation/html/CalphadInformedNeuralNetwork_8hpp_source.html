330 template <typename T>
331 void CalphadInformedNeuralNetwork<T>::initialize(
332  const std::vector<std::tuple<std::string, std::string>>& sorted_chemical_system) {
333  Catch_Time_Section("CalphadInformedNeuralNetwork<T>::initialize");
334  at::set_num_interop_threads(1);
335  at::set_num_threads(1);
336  // torch::AutoGradMode enable_grad(false);
338  // Check input composition consistency
339  for (int i = 0; i < sorted_chemical_system.size(); i++) {
340  const std::string& elem = std::get<0>(sorted_chemical_system[i]);
341  this->idx_elem_.emplace(elem, i);
342  }
343  if (!this->input_composition_order_.empty()) {
344  std::vector<std::string> v_elem;
345  for (const auto& tup : sorted_chemical_system) {
346  const std::string& elem = std::get<0>(tup);
347  v_elem.emplace_back(elem);
348  this->idx_elem_[elem] = std::distance(this->input_composition_order_.begin(),
349  std::find(this->input_composition_order_.begin(),
350  this->input_composition_order_.end(), elem));
351  }
352  // Sort before comparison
353  std::ranges::sort(v_elem);
354  auto sort_input_composition_order = this->input_composition_order_;
355  std::ranges::sort(sort_input_composition_order);
356 
357  MFEM_VERIFY(std::ranges::equal(v_elem, sort_input_composition_order),
358  "Error: InputCompositionOrder is not consistent with composition deduced from "
359  "auxiliary variables. Please check your data");
360  }
362  // Deserialize the ScriptModule for chemical potentials
363  if (!this->chemical_potentials_neural_network_.empty()) {
364  try {
365  for (const auto& [phase, mmodel] : this->chemical_potentials_neural_network_) {
366  // Deserialize the ScriptModule from a file using torch::jit::load()
367  this->chemical_potentials_nn_[phase] =
368  std::make_unique<torch::jit::script::Module>(torch::jit::load(mmodel));
369  }
370  } catch (const c10::Error& e) {
371  const std::string& error_msg =
372  "Error while loading neuronal network for chemical potentials. Please check the "
373  "input datafile";
374  mfem::mfem_error(error_msg.c_str());
375  }
376  }
377 
379  // Deserialize the ScriptModules for mobilities by phase
380  if (!this->mobilities_neural_network_.empty()) {
381  try {
382  for (const auto& [phase, mmodel] : this->mobilities_neural_network_) {
383  MFEM_VERIFY(phase != "LIQUID", "Error with mobility model deserialize ");
384 
385  // Deserialize the ScriptModule from a file using torch::jit::load()
386  this->mobilities_nn_[phase] =
387  std::make_unique<torch::jit::script::Module>(torch::jit::load(mmodel));
388  }
389  } catch (const c10::Error& e) {
390  const std::string& error_msg =
391  "Error while loading neuronal network for mobilities. Please check the "
392  "input datafile";
393  mfem::mfem_error(error_msg.c_str());
394  }
395  }
396 
398  // Deserialize the ScriptModules for mobilities by phase
399  if (!this->energies_neural_network_.empty()) {
400  try {
401  for (const auto& [phase, mmodel] : this->energies_neural_network_) {
402  // Deserialize the ScriptModule from a file using torch::jit::load()
403  this->energies_nn_[phase] =
404  std::make_unique<torch::jit::script::Module>(torch::jit::load(mmodel));
405  }
406  } catch (const c10::Error& e) {
407  const std::string& error_msg =
408  "Error while loading neuronal network for energies. Please check the "
409  "input datafile";
410  mfem::mfem_error(error_msg.c_str());
411  }
412  }

459 template <typename T>
460 void CalphadInformedNeuralNetwork<T>::compute(
461  const std::set<int>& list_nodes, const std::vector<T>& tp_gf,
462  const std::vector<std::tuple<std::string, std::string>>& chemical_system,
463  const std::string& given_phase) {
464  Catch_Time_Section("CalphadInformedNeuralNetwork<T>::compute");
465 
467  // Build tensor of inputs (all nodes)
469  const size_t nb_nodes = list_nodes.size();
470  std::vector<double> tp_gf_at_node(tp_gf.size());
471  // By default, the number of inputs is the number of chemical element + temperature
472  int nb_input = chemical_system.size() + 1;
473  // The number of inputs is reduced if the reference element is removed from inputs
474  // (consistent when working with molar fractions)
475  if (this->element_removed_from_inputs_) {
476  nb_input--;
477  }
478  // The number of inputs is enhanced if the pressure is an input of all meta-models
479  if (this->model_built_with_pressure_) {
480  nb_input++;
481  }
482 
483  auto options = torch::TensorOptions().dtype(torch::kDouble);
484  auto tensor =
485  torch::zeros({static_cast<int64_t>(nb_nodes), static_cast<int64_t>(nb_input)}, options);
486 
487  // ia is the index of tensor. ia is not equal to i because the list of nodes is not necessary from
488  // 0 to nb_nodes
489  int ia = -1;
490  for (const auto& i : list_nodes) {
491  ia++;
492  // Populate tp_gf_at_node for the current node
493  std::transform(tp_gf.begin(), tp_gf.end(), tp_gf_at_node.begin(),
494  [&i](const T& vec) { return vec[i]; });
495 
496  const double temperature = tp_gf_at_node[0];
497  const double pressure = tp_gf_at_node[1];
498 
499  tensor[ia][0] = temperature;
500 
501  // id_elem starts to 1 if pressure is not an input of the model. Otherwise it starts to 2.
502  int id_elem = 1;
503  if (this->model_built_with_pressure_) {
504  tensor[ia][1] = pressure;
505  id_elem = 2;
506  }
507  // sum is used to calculate the molar fraction of the element removed from the inputs
508  double sum_x = 0.0;
509  for (std::size_t iel = 0; iel < chemical_system.size(); ++iel) {
510  const std::string& elem = std::get<0>(chemical_system[iel]);
511  if (this->element_removed_from_inputs_ && elem == this->element_removed_from_nn_inputs_)
512  continue;
513 
514  int idx_el = this->idx_elem_[elem];
515  if ((this->element_removed_from_inputs_) &&
516  (this->idx_elem_[elem] > this->idx_elem_[this->element_removed_from_nn_inputs_])) {
517  idx_el = this->idx_elem_[elem] - 1;
518  }
519 
520  tensor[ia][idx_el + id_elem] = tp_gf_at_node[iel + 2] * input_composition_factor_;
521  const auto& key = std::make_tuple(i, given_phase, elem);
522  this->elem_mole_fraction_by_phase_[key] = tp_gf_at_node[iel + 2];
523  if (this->element_removed_from_inputs_) {
524  sum_x += this->elem_mole_fraction_by_phase_[key];
525  }
526  }
527  // Molar fraction of the removed element by phase
528  if (this->element_removed_from_inputs_) {
529  this->elem_mole_fraction_by_phase_[std::make_tuple(
530  i, given_phase, this->element_removed_from_nn_inputs_)] = 1.0 - sum_x;
531  }
532  }
533 
534  std::vector<torch::jit::IValue> inputs;
535  inputs.push_back(tensor);
536 
538  // Get outputs (all nodes)
540  // All tensors are detached in order to avoid the overhead of gradient calculation and then to
541  // save memory/CPU time
542  // Tensors are also expected stored in a contiguous block memory
543 
544  // Chemical potentials
545  auto output_mu_tmp = this->chemical_potentials_nn_[given_phase]->forward(inputs);
546  at::Tensor output_mu = output_mu_tmp.toTensor().detach();
547 
548  // Mobilities
549  at::Tensor output_mob_tensor;
550  if (this->has_nn_mobilities_ && this->mobilities_nn_.contains(given_phase)) {
551  output_mob_tensor = output_mu;
552  if (this->own_mobilities_model_) {
553  auto output_mob_tmp = this->mobilities_nn_[given_phase]->forward(inputs);
554  output_mob_tensor = output_mob_tmp.toTensor().detach();
555  }
556  if (!output_mob_tensor.is_contiguous()) {
557  output_mob_tensor = output_mob_tensor.contiguous();
558  }
559  }
560  // Energies
561  at::Tensor output_nrj_tensor;
562  if (this->has_nn_energies_ && this->energies_nn_.contains(given_phase)) {
563  output_nrj_tensor = output_mu;
564  if (this->own_energies_model_) {
565  auto output_nrj_tmp = this->energies_nn_[given_phase]->forward(inputs);
566  output_nrj_tensor = output_nrj_tmp.toTensor().detach();
567  }
568  if (!output_nrj_tensor.is_contiguous()) {
569  output_nrj_tensor = output_nrj_tensor.contiguous();
570  }
571  }
572 
573  if (!output_mu.is_contiguous()) {
574  output_mu = output_mu.contiguous();
575  }
576 
577  const int output_mu_size = output_mu.sizes()[1];
578  const double* output_data_mu = output_mu.data_ptr<double>();
579 
580  int output_mob_size;
581  double* output_data_mob;
582  if (this->has_nn_mobilities_ && this->mobilities_nn_.contains(given_phase)) {
583  output_mob_size = output_mob_tensor.sizes()[1];
584  output_data_mob = output_mob_tensor.data_ptr<double>();
585  }
586 
587  int output_nrj_size;
588  double* output_data_nrj;
589  if (this->has_nn_energies_ && this->energies_nn_.contains(given_phase)) {
590  output_nrj_size = output_nrj_tensor.sizes()[1];
591  output_data_nrj = output_nrj_tensor.data_ptr<double>();
592  }
593 
594  // Fill Calphad containers in order to update outputs of the Calphad problem
595  ia = -1;
596  for (const auto& i : list_nodes) {
597  ia++;
598  for (std::size_t iel = 0; iel < chemical_system.size(); ++iel) {
599  const std::string& elem = std::get<0>(chemical_system[iel]);
600 
601  // Chemical potentials
602  const int& mu_index = this->index_chemical_potentials_neural_network_.at(given_phase);
603  this->chemical_potentials_[std::make_tuple(i, elem)] =
604  output_data_mu[ia * output_mu_size + mu_index + this->idx_elem_[elem]];
605 
606  // Mobilities
607  if (this->has_nn_mobilities_ && this->mobilities_nn_.contains(given_phase)) {
608  const int& mob_index = this->index_mobilities_neural_network_.at(given_phase);
609  this->mobilities_[std::make_tuple(i, given_phase, elem)] =
610  output_data_mob[ia * output_mob_size + mob_index + this->idx_elem_[elem]];
611  }
612 
613  // Energies
614  if (this->has_nn_energies_ && this->energies_nn_.contains(given_phase)) {
615  // ir = index of energy_name
616  int ir = 0;
617  for (const auto& energy_name : this->input_energies_order_) {
618  const auto& key = std::make_tuple(i, given_phase, energy_name);
