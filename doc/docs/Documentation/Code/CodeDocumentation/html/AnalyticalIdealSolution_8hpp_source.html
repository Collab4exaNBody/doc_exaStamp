160 template <typename T>
161 void AnalyticalIdealSolution<T>::compute(
162  const std::set<int>& list_nodes, const std::vector<T>& tp_gf,
163  const std::vector<std::tuple<std::string, std::string>>& chemical_system) {
164  // Let us assume an ideal mixing solution
165  const std::string& phase = "SOLUTION";
166  const std::vector<std::string> energy_names = {"G"};
167  std::vector<double> tp_gf_at_node(tp_gf.size());
168 
169  const auto temperature_sort_method =
170  this->params_.template get_param_value_or_default<std::string>("temperature_sort_method",
171  "No");
172  const auto pressure_sort_method =
173  this->params_.template get_param_value_or_default<std::string>("pressure_sort_method", "No");
174 
175  std::vector<int> sorted_n_t_p =
176  this->CU_->sort_nodes(tp_gf[0], tp_gf[1], temperature_sort_method, pressure_sort_method);
177 
178  // Remove elements from list_nodes in sorted_n_t_p
179  // Usefull if the targeted list is lower than the complete list (eg. KKS problems)
180  sorted_n_t_p.erase(
181  std::remove_if(sorted_n_t_p.begin(), sorted_n_t_p.end(),
182  [&list_nodes](int node) { return list_nodes.find(node) == list_nodes.end(); }),
183  sorted_n_t_p.end());
184 
185  // Process CALPHAD calculations for each node
186  for (const auto& id : sorted_n_t_p) {
187  // Populate tp_gf_at_node for the current node
188  std::transform(tp_gf.begin(), tp_gf.end(), tp_gf_at_node.begin(),
189  [&id](const T& vec) { return vec[id]; });
190  const auto Temp = tp_gf_at_node[0];
191  const auto& epsilon = 1.e-10;
192  const auto x = std::max(epsilon, std::min(1. - epsilon, tp_gf_at_node[2]));
193  const auto& elem = std::get<0>(chemical_system[0]);
194 
195  // Energy
196  for (const auto& energy_name : energy_names) {
197  const auto& key = std::make_tuple(id, phase, energy_name);
198  this->energies_of_phases_[key] =
199  Physical::R * Temp * (x * std::log(x) + (1. - x) * std::log(1. - x));
200  }
201 
202  // Chemical potential
203  this->chemical_potentials_[std::make_tuple(id, elem)] =
204  Physical::R * Temp * (std::log(x) - std::log(1. - x));
205 
206  // Diffusion chemical potential
207  this->diffusion_chemical_potentials_[std::make_tuple(id, elem)] =
208  Physical::R * Temp * (std::log(x) - std::log(1. - x));
209 
210  // Molar fraction
211  const auto& key = std::make_tuple(id, phase, elem);
212  this->elem_mole_fraction_by_phase_[key] = x;
213  }
214 }
