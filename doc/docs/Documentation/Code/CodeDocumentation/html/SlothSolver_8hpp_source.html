185  template <typename Solv, typename Prec>
186  inline void operator()(Solv&& solv, Prec&& prec) {
187  using TT = std::decay_t<decltype(solv)>;
188  using PP = std::decay_t<decltype(prec)>;
189  UtilsSolvers::print_solver_name(solv);
190  UtilsSolvers::print_prec_name(prec);
191  if constexpr (!std::is_same_v<PP, sptr<std::monostate>>) {
192  if constexpr (is_in_variant_v<TT, VIterativeSolver>) {
193  MFEM_VERIFY((is_in_variant_v<PP, VIterativePrecond>),
194  "SetPrecondSolver: IterativeSolver objects can only be associated with an "
195  "IterativePreconditionner objects");
196  if constexpr (is_in_variant_v<PP, VIterativePrecond>) {
197  SlothInfo::debug("SetPrecondSolver: setting iterative preconditionner");
198  UtilsSolvers::setter_mfem(solv, prec);
199  }
200  }
201  if constexpr (is_in_variant_v<TT, VHypreSolver>) {
202  MFEM_VERIFY((is_in_variant_v<PP, VHyprePrecond>),
203  "SetPrecondSolver: HypreSolver objects can only be associated with an "
204  "HyprePreconditionner objects");
205  if constexpr (is_in_variant_v<PP, VHyprePrecond>) {
206  SlothInfo::debug("SetPrecondSolver: setting hypre preconditionner (not hypre smoother)");
207  UtilsSolvers::setter_hypre(solv, prec);
208  }
209  }
210  }
211 
212  if constexpr (!std::is_same_v<TT, sptr<std::monostate>>) {
213  SlothInfo::debug("SetPrecondSolver: setting operator ");
214  solv->SetOperator(this->op);
215  }
216  }
