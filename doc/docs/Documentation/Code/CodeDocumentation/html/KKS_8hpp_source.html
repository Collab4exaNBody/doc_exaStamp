233 template <typename T>
234 std::set<int> KKS<T>::check_nucleation(CalphadBase<T> &CALPHAD, const std::set<int> &indices_ph_1,
235  const T &tp_gf_ph_1) {
236  std::set<int> indices_nucleation;
237 
238  switch (KKS_nucleation_strategy::from(this->KKS_nucleation_strategy_)) {
239  //
240  case KKS_nucleation_strategy::liquid_fraction: {
241  for (const auto &node : indices_ph_1) {
242  // Check only if equilibrium is found
243  if (CALPHAD.error_equilibrium_[node] == CalphadDefaultConstant::error_max) continue;
244  // Check if secondary phase is found
245  if (CALPHAD.elem_mole_fraction_by_phase_.contains(std::make_tuple(
246  node, this->KKS_secondary_phase_, this->element_removed_from_ic_))) {
247  indices_nucleation.insert(node);
248  }
249  }
250  break;
251  }
252  //
253  case KKS_nucleation_strategy::given_melting_temperature: {
254  for (const auto &node : indices_ph_1) {
255  // Check if temperature at node is greater than a given limit
256  if (tp_gf_ph_1(node) > this->given_melting_temperature_) {
257  indices_nucleation.insert(node);
258  }
259  }
260  break;
261  }
262  default: {
263  throw std::runtime_error(
264  "KKS<T>::check_nucleation_at_node: error in the choice of method used to compute "
265  "the nucleation. Available choices are : liquid_fraction (Default), "
266  "given_melting_temperature");
267  }
268  }
269  return indices_nucleation;
270 }

312  if (phi > 1 - this->KKS_threshold_) {
313  indices_ph_1.insert(i);
314  } else if (phi < this->KKS_threshold_) {
315  SlothInfo::debug("Execute_linearization: secondary_phase node ", i);
316  indices_ph_2.insert(i);
317

333  MPI_Allreduce(&local_size, &global_size, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);

339  // Lambda for equilibrium calculations performed in the pure phase or in the interface with a
340  // temperature deviation
341  auto calculate_interface =
342  [&](const std::set<int> &indices, const std::vector<T> &delta_tp_gf, const double increment,
343  const int id_incr,
344  std::map<std::tuple<int, std::string, std::string>, double> &chemical_potential_interface,
345  std::vector<std::tuple<std::string, std::string, double>> given_phase,
346  const std::string &primary_phase) {
347  std::vector<T> delta_tp = delta_tp_gf;
348  if (id_incr > -1) {
349  delta_tp[id_incr] += increment;
350  }
351  CALPHAD.execute(dt, indices, delta_tp, chemicalsystem, given_phase);
352 
353  for (const auto &in : indices) {
354  if (CALPHAD.error_equilibrium_[in] == CalphadDefaultConstant::error_max) continue;
355  for (const auto &elem : chemicalsystem) {
356  const auto &[elem1, unit] = elem;
357  const double mu = CALPHAD.chemical_potentials_.at(std::make_tuple(in, elem1));
358  chemical_potential_interface.emplace(std::make_tuple(in, elem1, primary_phase), mu);
359  }
360  }
361  };

382  const double mu = CALPHAD.chemical_potentials_.at(std::make_tuple(in, elem1));

590 
591  // x - dx
592  // Equilibrium calculations in both phases at (Tbar, Xbar - delta X)
593  calculate_interface_x(bar_tp_gf_ph_1, -this->KKS_composition_increment_, ielem + 2, ielem,
594  this->chemical_potentials_left_x_, st_phase_1, phase);
595 
596  calculate_interface_x(bar_tp_gf_ph_2, -this->KKS_composition_increment_, ielem + 2, ielem,
597  this->chemical_potentials_left_x_, st_phase_2,
598  this->KKS_secondary_phase_);
599  }
600  ielem++;
601  }
602 
606  // Allocate
607  mfem::Array<int> offsets(3);
608  offsets[0] = 0;
609  offsets[1] = nb_elem - 1;
610  offsets[2] = 2 * nb_elem - 2;
611 
612  mfem::BlockVector bb(offsets);
613  mfem::BlockVector deltaX_phase(offsets);
614  mfem::Vector HphiNode(nb_elem - 1);
615  mfem::Vector one_minus_HphiNode(nb_elem - 1);
616 
617  std::unique_ptr<mfem::BlockMatrix> AA = std::make_unique<mfem::BlockMatrix>(offsets, offsets);
618 
619  auto solver = std::make_shared<mfem::GMRESSolver>();
620  solver->SetOperator(*AA);
621  solver->SetAbsTol(this->kks_abs_tol_solver_);
622  solver->SetRelTol(this->kks_rel_tol_solver_);
623  solver->SetMaxIter(this->kks_max_iter_solver_);
624  solver->SetPrintLevel(this->kks_print_level_solver_);
625 
626  for (const auto &node : indices_inter) {
627  // Solve KKS only if an equilibrium is found. Otherwise, solution at previous time-step will
628  // be taken into account
629  if (CALPHAD.error_equilibrium_[node] == CalphadDefaultConstant::error_max) continue;
630 
631  //
632  // Build system
633  //
634  // Matrix for primary and secondary phase
635 
636  std::unique_ptr<mfem::SparseMatrix> Al =
637  this->get_A4linearKKS(chemicalsystem, this->KKS_secondary_phase_, node);
638  std::unique_ptr<mfem::SparseMatrix> As = this->get_A4linearKKS(chemicalsystem, phase, node);
639  // Vector for primary and secondary phase
640  mfem::Vector hs = this->get_h4linearKKS(chemicalsystem, phase, node);
641  mfem::Vector hl = this->get_h4linearKKS(chemicalsystem, this->KKS_secondary_phase_, node);
642  mfem::Vector ms = this->get_m4linearKKS(chemicalsystem, phase, node);
643  mfem::Vector ml = this->get_m4linearKKS(chemicalsystem, this->KKS_secondary_phase_, node);
644 
645  // ml - ms + (T - Tbar) * (hl - hs)
646  // ml - ms
647  mfem::Vector ml_minus_ms(ml);
648  ml_minus_ms -= ms;
649 
650  // T - Tbar
651  const double deltaT = tp_gf[0](node) - bar_tp_gf_ph_1[0](node);
652 
653  // hl * (T - Tbar)
654  mfem::Vector hl_deltaT(hl);
655  hl_deltaT *= deltaT;
656 
657  // hs * (T - Tbar)
658  mfem::Vector hs_deltaT(hs);
659  hs_deltaT *= deltaT;
660 
661  // hl * (T - Tbar) - hs * (T - Tbar)
662  mfem::Vector hl_minus_hs(hl_deltaT);
663  hl_minus_hs -= hs_deltaT;
664 
665  // LHS : deltaX
666  mfem::Vector deltaX(nb_elem - 1);
667  int ielem = 0;
668  int jelem = 0;
669  for (const auto &[elem, unit] : chemicalsystem) {
670  if (elem != this->element_removed_from_ic_) {
671  const double current_x = tp_gf[jelem + 2](node);
672  const double old_x = tp_gf_old[jelem + 2](node);
673  const double dx = current_x - old_x;
674 
675  const double current_Hphi = Hphi(node);
676  const double old_Hphi = Hphi_old(node);
677 
678  const double xdH = (bar_tp_gf_ph_1[jelem + 2](node) - bar_tp_gf_ph_2[jelem + 2](node)) *
679  (current_Hphi - old_Hphi);
680 
681  SlothInfo::debug("Calculation of dx at node ", node, " = ", dx, " for elem ", elem,
682  " with x ", current_x, " and x_old ", old_x);
683  SlothInfo::debug("Calculation of xdH at node ", node, " = ", xdH, " for elem ", elem,
684  " with xs_old ", bar_tp_gf_ph_1[jelem + 2](node), " xl_old ",
685  bar_tp_gf_ph_2[jelem + 2](node), " Hphi(node) ", current_Hphi,
686  " and Hphi_old(node) ", old_Hphi);
687 
688  deltaX(ielem) = dx - xdH;
689 
690  ielem++;
691  }
692  jelem++;
693  }
694 
695  //
696  // Assemble system
697  //
698 
699  // mfem::BlockVector bb(offsets);
700  mfem::Vector &b0 = bb.GetBlock(0);
701  b0 = ml_minus_ms;
702  b0 += hl_minus_hs;
703 
704  mfem::Vector &b1 = bb.GetBlock(1);
705  b1 = deltaX;
706  mfem::Vector &d0 = deltaX_phase.GetBlock(0);
707  mfem::Vector &d1 = deltaX_phase.GetBlock(1);
708  d0 = 0.;
709  d1 = 0.;
710 
711  // mfem::BlockMatrix *AA = new mfem::BlockMatrix(offsets, offsets);
712  AA->SetBlock(0, 0, As.get());
713  *Al *= -1.;
714  AA->SetBlock(0, 1, Al.get());
715 
716  // identity
717  HphiNode = Hphi(node);
718  one_minus_HphiNode = 1. - Hphi(node);
719 
720  auto A10 = std::make_unique<mfem::SparseMatrix>(HphiNode);
721  auto A11 = std::make_unique<mfem::SparseMatrix>(one_minus_HphiNode);
722 
723  AA->SetBlock(1, 0, A10.get());
724  AA->SetBlock(1, 1, A11.get());
725 
726  AA->Finalize();
727 
728  //
729  // Solve system
730  //
731  // mfem::BlockVector deltaX_phase(offsets);
732  // Check result of linear system
733  if (Verbosity::Debug <= verbosityLevel) {
734  SlothInfo::print("BEfore solving, KKS linear system at node (A X = B):", node);
735  AA->Print();
736  deltaX_phase.Print();
737  bb.Print();
738  }
739 
740  solver->Mult(bb, deltaX_phase);
741  // Check result of linear system
742  if (Verbosity::Debug <= verbosityLevel) {
743  SlothInfo::print("KKS linear system at node (A X = B):", node);
744  AA->Print();
745  deltaX_phase.Print();
746  bb.Print();
747  }
748 
749  // Back to Al. Required before recovering thermodynamic contribution
750  *Al *= -1.;
751 
755  const mfem::Vector delta_XS = deltaX_phase.GetBlock(0);
756  const mfem::Vector delta_XL = deltaX_phase.GetBlock(1);
758  // Composition by phase
760  int i = 0;
761  int ie = 0;
762  double sumS = 0.;
763  double sumL = 0.;
764  for (const auto &[elem, unit] : chemicalsystem) {
765  if (elem != this->element_removed_from_ic_) {
766  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(node, phase, elem)] =
767  bar_tp_gf_ph_1[i + 2](node) + delta_XS(ie);
768  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
769  node, this->KKS_secondary_phase_, elem)] = bar_tp_gf_ph_2[i + 2](node) + delta_XL(ie);
770  sumS += CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(node, phase, elem)];
771  sumL += CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
772  node, this->KKS_secondary_phase_, elem)];
773  ie++;
774  }
775  ++i;
776  }
777  // Last component in each phase
778  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
779  node, phase, this->element_removed_from_ic_)] = 1. - sumS;
780  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
781  node, this->KKS_secondary_phase_, this->element_removed_from_ic_)] = 1. - sumL;
782 
784  // Difference of chemical potentials
786  mfem::Vector dmu_s(ms);
787  dmu_s += hs_deltaT;
788 
789  mfem::Vector AsDeltaXs(nb_elem - 1);
790  As->Mult(delta_XS, AsDeltaXs);
791  dmu_s += AsDeltaXs;
792 
793  mfem::Vector dmu_l(ml);
794  dmu_l += hl_deltaT;
795  mfem::Vector AlDeltaXl(nb_elem - 1);
796  Al->Mult(delta_XL, AlDeltaXl);
797  dmu_l += AlDeltaXl;
798 
800  // Gibbs energies (implicit case)
802  double gs = CALPHAD.energies_of_phases_[std::make_tuple(node, phase, "GM")];
803  double gl =
804  CALPHAD.energies_of_phases_[std::make_tuple(node, this->KKS_secondary_phase_, "GM")];
805  auto DeltaXsAsDeltaXs = AsDeltaXs * delta_XS;
806  DeltaXsAsDeltaXs *= 0.5;
807  auto DeltaXlAlDeltaXl = AlDeltaXl * delta_XL;
808  DeltaXlAlDeltaXl *= 0.5;
809  auto DeltaXsMs = ms * delta_XS;
810  auto DeltaXlMl = ml * delta_XL;
811  gs += DeltaXsMs;
812  gs += DeltaXsAsDeltaXs;
813  gl += DeltaXlMl;
814  gl += DeltaXlAlDeltaXl;
815 
816  // Molar Gibbs Energy by phase
817  CALPHAD.energies_of_phases_[std::make_tuple(node, phase, "GM")] = gs;
818  CALPHAD.energies_of_phases_[std::make_tuple(node, this->KKS_secondary_phase_, "GM")] = gl;
819 
820  // Driving force by phase : contribution -gm
821  CALPHAD.driving_forces_[std::make_tuple(node, phase)] =
822  -CALPHAD.energies_of_phases_[std::make_tuple(node, phase, "GM")];
823  CALPHAD.driving_forces_[std::make_tuple(node, this->KKS_secondary_phase_)] =
824  -CALPHAD.energies_of_phases_[std::make_tuple(node, this->KKS_secondary_phase_, "GM")];
825 
826  i = 0;
827  ie = 0;
828  for (const auto &[elem, unit] : chemicalsystem) {
829  if (elem != this->element_removed_from_ic_) {
830  SlothInfo::debug("Check at node ", node, " for elem ", elem);
831  SlothInfo::debug("deltaX to obtain ", deltaX(ie), " deltaX predicted ",
832  delta_XS(ie) * Hphi(node) + delta_XL(ie) * (1. - Hphi(node)));
833  SlothInfo::debug(
834  "x to obtain ", tp_gf[i + 2](node), " x predicted ",
835  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(node, phase, elem)] *
836  Hphi(node) +
837  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(
838  node, this->KKS_secondary_phase_, elem)] *
839  (1. - Hphi(node)));
840 
841  SlothInfo::debug("Diffusion chemical potential: primary phase ", dmu_s(ie),
842  " ; secondary phase", dmu_l(ie));
843 
844  // Diffusion chemical potential
845  CALPHAD.diffusion_chemical_potentials_[std::make_tuple(node, elem)] = dmu_s(ie);
846 
847  // Pseudo Driving force : contribution dmu_i * x_i
848  //
849  CALPHAD.driving_forces_[std::make_tuple(node, phase)] +=
850  CALPHAD.diffusion_chemical_potentials_[std::make_tuple(node, elem)] *
851  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(node, phase, elem)];
852 
853  CALPHAD.driving_forces_[std::make_tuple(node, this->KKS_secondary_phase_)] +=
854  CALPHAD.diffusion_chemical_potentials_[std::make_tuple(node, elem)] *
855  CALPHAD.elem_mole_fraction_by_phase_[std::make_tuple(node, this->KKS_secondary_phase_,
856  elem)];
857  ie++;
858  }
859  i++;
860  }
861 
862  SlothInfo::debug("Gibbs energy: primary phase ", gs, " ; secondary phase ", gl);
863  SlothInfo::debug("DGM: primary phase ", CALPHAD.driving_forces_[std::make_tuple(node, phase)],
864  " ; secondary phase ",
865  CALPHAD.driving_forces_[std::make_tuple(node, this->KKS_secondary_phase_)]);
866  }
867  }
868 }

879 template <typename T>
880 std::unique_ptr<mfem::SparseMatrix> KKS<T>::get_A4linearKKS(
881  const std::vector<std::tuple<std::string, std::string>> &chemicalsystem,
882  const std::string &phase, const int node) {
883  const int nb_elem = chemicalsystem.size();
884  auto AA = std::make_unique<mfem::SparseMatrix>(nb_elem - 1, nb_elem - 1);
885 
886  // Diagonal
887  // d2dxdx = d(mu_x - mu-n)/dx =[ (mu_x - mu-n)(x+dx) - (mu_x - mu-n)(x-dx) ] / 2dx
888  // elem_id corresponds to index of variable with delta_x, index of derivative
889  int elem_id = 0;
890  int vid = 0;
891  for (const auto &[elem, unit] : chemicalsystem) {
892  if (elem != this->element_removed_from_ic_) {
893  double d2gd2x =
894  this->chemical_potentials_right_x_[std::make_tuple(elem_id, node, elem, phase)] -
895  this->chemical_potentials_right_x_[std::make_tuple(
896  elem_id, node, this->element_removed_from_ic_, phase)] -
897  (this->chemical_potentials_left_x_[std::make_tuple(elem_id, node, elem, phase)] -
898  this->chemical_potentials_left_x_[std::make_tuple(
899  elem_id, node, this->element_removed_from_ic_, phase)]);
900  d2gd2x /= 2. * this->KKS_composition_increment_;
901  AA->Set(vid, vid, d2gd2x);
902  vid++;
903  }
904  elem_id++;
905  }
906  // Off-diagonal (and symmetry)
907  elem_id = 0;
908  vid = 0;
909  for (const auto &[ielem, unit] : chemicalsystem) {
910  if (ielem != this->element_removed_from_ic_) {
911  int elem_jd = 0;
912  int vjd = 0;
914  // row i
915  for (const auto &[jelem, unit] : chemicalsystem) {
916  if (jelem != this->element_removed_from_ic_) {
918  // col j
919  //
920  // d2dxdy = 0.5 d(mu_x - mu-n)/dy + 0.5 d(mu_y - mu-n)/dx
921  if (ielem != jelem) {
922  double dmuxdy =
923  (this->chemical_potentials_right_x_[std::make_tuple(elem_jd, node, ielem, phase)] -
924  this->chemical_potentials_right_x_[std::make_tuple(
925  elem_jd, node, this->element_removed_from_ic_, phase)]) -
926  ((this->chemical_potentials_left_x_[std::make_tuple(elem_jd, node, ielem, phase)] -
927  this->chemical_potentials_left_x_[std::make_tuple(
928  elem_jd, node, this->element_removed_from_ic_, phase)]));
929  dmuxdy /= 2. * this->KKS_composition_increment_;
930  double dmuydx =
931  (this->chemical_potentials_right_x_[std::make_tuple(elem_id, node, jelem, phase)] -
932  this->chemical_potentials_right_x_[std::make_tuple(
933  elem_id, node, this->element_removed_from_ic_, phase)]) -
934  ((this->chemical_potentials_left_x_[std::make_tuple(elem_id, node, jelem, phase)] -
935  this->chemical_potentials_left_x_[std::make_tuple(
936  elem_id, node, this->element_removed_from_ic_, phase)]));
937  dmuydx /= 2. * this->KKS_composition_increment_;
938 
939  double d2gdxdy = 0.5 * dmuxdy + 0.5 * dmuydx;
940 
941  AA->Set(vid, vjd, d2gdxdy);
942  }
943  vjd++;
945  }
946  elem_jd++;
947  }
949  vid++;
950  }
951  elem_id++;
952  }
953 
954  return AA;
955 }
