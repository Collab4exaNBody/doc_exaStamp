142  if (calphad_outputs::from(symbol) != calphad_outputs::dgm) continue;

213 double AllenCahnCalphadMeltingNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI, INTERPOLATION>::
214  get_phase_change_at_ip(mfem::ElementTransformation& Tr, const mfem::IntegrationPoint& ir) {
215  double primary_dgm = -1.;
216  double secondary_dgm = -1.;
217  if (this->dgm_.size() == 2) {
218  primary_dgm = this->dgm_[0].GetValue(Tr, ir);
219  secondary_dgm = this->dgm_[1].GetValue(Tr, ir);
220  }
221  constexpr double epsilon = 1.e-12;
222  return (std::abs(primary_dgm) > epsilon && std::abs(secondary_dgm) > epsilon)
223  ? (this->alpha_ * (secondary_dgm - primary_dgm))
224  : 0.;
225 }
