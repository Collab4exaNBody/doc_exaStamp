104 void ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency() {
105  std::vector<mfem::ParGridFunction> aux_gf = this->get_aux_gf();
106  std::vector<std::vector<std::string>> aux_infos = this->get_aux_infos();
107 
108  //==============================================================
109  // Get Temperature and thermal diffusivity (aux. variables)
110  //==============================================================
111  bool temperature_found = false;
112  bool diffusivity_found = false;
113  for (std::size_t i = 0; i < aux_infos.size(); ++i) {
114  const auto& variable_info = aux_infos[i];
115 
116  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
117 
118  size_t vsize = variable_info.size();
119 
120  MFEM_VERIFY(vsize > 1,
121  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: at least "
122  "two additionnal information are expected for auxiliary variables associated with "
123  "this integrator");
124  const std::string& symbol = toUpperCase(variable_info.back());
125 
126  if (toUpperCase(symbol) == "T") {
127  this->temp_gf_ = aux_gf[i];
128  temperature_found = true;
129  } else if (toLowerCase(symbol) == "lambda") {
130  // Diffusivity can be directly supplied within this integrator or overloaded by considering a
131  // child class.
132  this->diffu_gf_ = aux_gf[i];
133  this->diffusivity_found_ = true;
134  }
135 
136  if (temperature_found && diffusivity_found) break;
137  }
138 
139  MFEM_VERIFY(temperature_found,
140  "ThermalDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
141  "Temperature not found. Please check your data.");
142 }
