230 FType AllenCahnNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::double_well_derivative(
231  const int order_derivative, mfem::ElementTransformation& Tr, const mfem::IntegrationPoint& ir) {
232  return FType([this, order_derivative, &Tr, &ir](const double& u) {
233  const auto& un = this->u_old_[0].GetValue(Tr, ir);
234  FType W_derivative;
235  if (order_derivative == 1) {
236  W_derivative = this->energy_first_derivative_potential_.getPotentialFunction(un);
237  } else if (order_derivative == 2) {
238  W_derivative = this->energy_second_derivative_potential_.getPotentialFunction(un);
239  } else {
240  std::runtime_error("Error while setting the order of derivative : only 1 and 2 are allowed.");
241  }
242  const double omega = this->omega(Tr, ir, u, this->params_);
243 
244  const auto& w_prime = omega * W_derivative(u);
245  return w_prime;
246  });
247 }

279 void AllenCahnNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::check_variables_consistency() {
280  this->aux_gf_ = this->get_aux_gf();
281  this->aux_old_gf_ = this->get_aux_old_gf();
282  this->aux_gf_infos_ = this->get_aux_infos();
283 
284  // Temperature scaling for mobility
285  bool temperature_found = false;
286  for (std::size_t i = 0; i < this->aux_gf_infos_.size(); ++i) {
287  const auto& variable_info = this->aux_gf_infos_[i];
288  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
289  size_t vsize = variable_info.size();
290 
291  MFEM_VERIFY(vsize >= 1,
292  "AllenCahnNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>: at least "
293  "one additionnal information is expected for auxiliary variables associated with "
294  "this integrator");
295  const std::string& symbol = toUpperCase(variable_info.back());
296  if (symbol == "T") {
297  this->temp_gf_.emplace_back(std::move(this->aux_gf_[i]));
298  temperature_found = true;
299  break;
300  }
301  }
302  if (this->params_.has_parameter("ScaleMobilityByTemperature")) {
303  this->scale_mobility_by_temperature_ =
304  this->params_.template get_param_value<bool>("ScaleMobilityByTemperature");
305  if (this->scale_mobility_by_temperature_) {
306  MFEM_VERIFY(
307  temperature_found,
308  "AllenCahnNLFormIntegrator: "
309  "Temperature variable required to scale mobility, but not found in auxiliary variables");
310  }
311  }
312 }

327 void AllenCahnNLFormIntegrator<VARS, SCHEME, ENERGY, MOBI>::AssembleElementVector(
328  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
329  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
330  int num_blocks = el.Size();
331  for (int blk = 0; blk < num_blocks; ++blk) {
332  // Catch_Time_Section("AllenCahnNLFormIntegrator:AssembleElementVector");
333  int nd = el[blk]->GetDof();
334  int dim = el[blk]->GetDim();
335  gradPsi.SetSize(nd, dim);
336  Psi.SetSize(nd);
337  gradU.SetSize(dim);
338  // elvect.SetSize(nd);
339  elvect[blk]->SetSize(nd);
340  *elvect[blk] = 0.;
341 
342  const mfem::IntegrationRule* ir =
343  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
344  // elvect = 0.0;
345  for (int i = 0; i < ir->GetNPoints(); i++) {
346  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
347  el[blk]->CalcShape(ip, Psi); //
348  Tr.SetIntPoint(&ip);
349 
350  const auto& u = *elfun[blk] * Psi;
351 
352  // Laplacian : given u, compute (grad(u), grad(psi)), psi is shape function.
353  // given u (elfun), compute grad(u)
354  el[blk]->CalcPhysDShape(Tr, gradPsi);
355  gradPsi.MultTranspose(*elfun[blk], gradU);
356  const double coef_mob = this->mobility(Tr, ip, u, this->params_) * ip.weight * Tr.Weight();
357  const double lambda = this->lambda(Tr, ip, u, this->params_);
358  gradU *= coef_mob * lambda;
359  gradPsi.AddMult(gradU, *elvect[blk]);
360 
361  // Given u, compute (w'(u), psi), psi is shape function
362  const double ww = coef_mob * this->energy_derivatives(1, Tr, ip)(u);
363  add(*elvect[blk], ww, Psi, *elvect[blk]);
364  }
365  }
366 }
