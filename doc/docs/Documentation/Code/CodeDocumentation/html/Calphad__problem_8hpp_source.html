385 template <class CALPHAD, class VAR, class PST>
386 void Calphad_Problem<CALPHAD, VAR, PST>::check_variables_consistency() {
387  // ---------------------
388  // Primary variables
389  // ---------------------
390  for (const auto& var : this->variables_.getVariables()) {
391  MFEM_VERIFY(var.get_additional_variable_info().size() > 0,
392  "Calphad problems requires ouputs with at least two additional informations \n");
393 
394  const std::string& symbol = toLowerCase(var.get_additional_variable_info().back());
395  switch (calphad_outputs::from(symbol)) {
396  case calphad_outputs::mu: {
397  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
398  "Calphad problems requires that chemical potential ouputs are defined with two "
399  "additional informations: first an element, second the symbol 'mu'. \n");
400 
401  SlothInfo::debug("Output : chemical potential for ", var.get_additional_variable_info()[0]);
402  break;
403  }
404  case calphad_outputs::dmu: {
405  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
406  "Calphad problems requires that diffusion chemical potential ouputs are "
407  "defined with two "
408  "additional informations: first an element and second the symbol 'dmu'. \n");
409 
410  SlothInfo::debug("Output : diffusion chemical potential for ",
411  var.get_additional_variable_info()[0]);
412  break;
413  }
414  case calphad_outputs::xph: {
415  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
416  "Calphad problems requires that phase molar fraction ouputs are defined with "
417  "two additional informations: first a phase and second the symbol 'xph'. \n");
418  SlothInfo::debug("Output : molar fraction for ", var.get_additional_variable_info()[0]);
419  break;
420  }
421  case calphad_outputs::xp: {
422  MFEM_VERIFY(var.get_additional_variable_info().size() == 3,
423  "Calphad problems requires that element molar fraction ouputs are defined with "
424  "three additional informations: first an element, second a phase and third "
425  "the symbol 'xp'. \n");
426  SlothInfo::debug("Output : molar fraction for ", var.get_additional_variable_info()[0]);
427  break;
428  }
429  case calphad_outputs::g:
430  case calphad_outputs::gm: {
431  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
432  "Calphad problems requires that Gibbs energy ouputs are defined with two "
433  "additional informations: first a phase, second the symbol 'g' or 'gm'. \n");
434  SlothInfo::debug("Output : Gibbs energy for ", var.get_additional_variable_info()[0]);
435  break;
436  }
437  case calphad_outputs::h:
438  case calphad_outputs::hm: {
439  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
440  "Calphad problems requires that enthalpy ouputs are defined with two "
441  "additional informations: first a phase, second the symbol 'h' or 'hm'. \n");
442  SlothInfo::debug("Output : enthalpy for ", var.get_additional_variable_info()[0]);
443  break;
444  }
445  case calphad_outputs::dgm: {
446  MFEM_VERIFY(var.get_additional_variable_info().size() == 2,
447  "Calphad problems requires that driving forces ouputs are defined with the "
448  "name of the phase and the symbol 'dgm' as the additional information. \n");
449  SlothInfo::debug("Output : driving force for ", var.get_additional_variable_info()[0]);
450  break;
451  }
452  case calphad_outputs::mob: {
453  MFEM_VERIFY(
454  var.get_additional_variable_info().size() == 3,
455  "Calphad problems requires that mobility ouputs are defined with two "
456  "additional informations: first a phase, second an element, the symbol 'mob'. \n");
457 
458  SlothInfo::debug("Output : mobility for ", var.get_additional_variable_info()[0]);
459  break;
460  }
461  case calphad_outputs::cp: {
462  MFEM_VERIFY(var.get_additional_variable_info().size() == 1,
463  "Calphad problems requires that heat capacity ouputs are defined with the "
464  "symbol cp as the only additional information. \n");
465  SlothInfo::debug("Output : heat capacity ");
466  break;
467  }
468  }
469  }
470  // ---------------------
471  // Auxiliary variables
472  // ---------------------
473  MFEM_VERIFY(this->auxvariables_.size() > 2,
474  "Calphad problems requires three set of auxiliary variables: the temperature, the "
475  "pressure and the composition");
476  // Temperature and Pressure
477  bool has_temperature = false;
478  bool has_pressure = false;
479  for (const auto& auxvar_vec : this->auxvariables_) {
480  for (const auto& auxvar : auxvar_vec->getVariables()) {
481  const std::string& symbol = toUpperCase(auxvar.get_additional_variable_info().back());
482  if (symbol == "T") {
483  has_temperature = true;
484  }
485  if (symbol == "P") {
486  has_pressure = true;
487  }
488  }
489  }
490  MFEM_VERIFY(has_temperature && has_pressure,
491  "Calphad problems requires temperature and pressure as auxiliary variables");
492 
493  // Composition
494  // TODO(cci) : add check of the type of variable among allowed type
495 }

524 template <class CALPHAD, class VAR, class PST>
525 void Calphad_Problem<CALPHAD, VAR, PST>::do_time_step(
526  double& next_time, const double& current_time, double current_time_step, const int iter,
527  std::vector<std::unique_ptr<mfem::Vector>>& vect_unk,
528  const std::vector<std::vector<std::string>>& unks_info) {
529  std::vector<mfem::Vector> tp_gf = this->get_tp_conditions();
530  std::vector<std::tuple<std::vector<std::string>, std::reference_wrapper<mfem::Vector>>>
531  output_system = this->get_output_system(unks_info, vect_unk);
532 
533  std::vector<std::tuple<std::vector<std::string>, mfem::Vector>> previous_output =
534  get_previous_output_system();
535 
536  const size_t unk_size = vect_unk.size();
537 
538  // Execute
539  if (!this->is_KKS_) {
540  // Calculate equilibrium at each node of the mesh without KKS algorithm
541  this->CC_->global_execute(iter, current_time_step, tp_gf, this->sorted_chemical_system_,
542  output_system, previous_output);
543  } else {
544  // Specific treatment for KKS problems
545  std::tuple<std::string, mfem::Vector, mfem::Vector> phasefields_gf = this->get_phasefields();
546  // Coordinates of each nodes to assess the radius of nucleii
547  std::vector<std::tuple<std::string, mfem::Vector>> coordinates_gf = this->get_coordinates();
548  // initial conditions at the previous time-step
549  std::vector<mfem::Vector> tp_gf_old = this->get_old_tp_conditions();
550  // Molar fractions by phase at current and previous time-step : vector<element, phase, x, x_old>
551  std::vector<std::tuple<std::string, std::string, mfem::Vector, mfem::Vector>> x_phase_gf =
552  this->get_molar_fractions();
553 
554  // Calculate equilibrium at each node of the mesh with KKS algorithm
555  this->CC_->global_execute(iter, current_time_step, tp_gf, this->sorted_chemical_system_,
556  output_system, previous_output, phasefields_gf, tp_gf_old, x_phase_gf,
557  coordinates_gf);
558  }
559 
560  // Recover unknowns
561  for (size_t i = 0; i < unk_size; i++) {
562  auto& unk_i = *(vect_unk[i]);
563  this->unknown_.emplace_back(unk_i);
564  }
565  // Update time
566  next_time = current_time + current_time_step;
567 }

646  for (const auto& auxvar : auxvar_vec->getVariables()) {

649  auto variable_info = auxvar.get_additional_variable_info();

654  "Error while getting phase_field. Expected [name of the phase, 'phi']");
655  aux_gf = std::make_tuple(variable_info[0], gf, gf_old);
656  }
657  }

735

750  "[component, 'x'] or [component, 'N']");

752 
753  aux_gf[id + 2] = gf;
754  }
755  }
756  }
757 
758  return aux_gf;
759 }
