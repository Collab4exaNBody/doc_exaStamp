348 template <class OPE, class VAR, class PST>
349 void Problem<OPE, VAR, PST>::post_processing(const int& iter, const double& current_time,
350  const double& current_time_step) {
351  const auto nvars = this->variables_.get_variables_number();
352  std::vector<mfem::Vector> u_vect;
353 
354  // Isovalue to compute by variable
355  const std::map<std::string, double> map_iso_value = this->pst_.get_iso_val_to_compute();
356 
357  // Integral value to compute by variable
358  const std::map<std::string, std::tuple<double, double>> map_integral =
359  this->pst_.get_integral_to_compute();
360 
361  for (auto iv = 0; iv < nvars; iv++) {
362  auto vv = this->variables_.getIVariable(iv);
363  auto solution = vv.get_analytical_solution();
364  auto unk = vv.get_unknown();
365  auto unk_name = vv.getVariableName();
366 
367  // Errors
368  if (solution != nullptr) {
369  auto solution_func = solution.get();
370  this->oper_.ComputeError(iter, current_time, current_time_step, iv, unk_name, unk,
371  *solution_func);
372  }
373 
374  // Isovalues
375  if (!map_iso_value.empty() && map_iso_value.contains(unk_name)) {
376  const double iso_value = map_iso_value.at(unk_name);
377  this->oper_.ComputeIsoVal(iter, current_time, current_time_step, iv, unk_name, unk,
378  iso_value);
379  }
380 
381  // Integral
382  if (!map_integral.empty() && map_integral.contains(unk_name)) {
383  const auto& [lower_bound, upper_bound] = map_integral.at(unk_name);
384  this->oper_.ComputeIntegral(iter, current_time, current_time_step, iv, unk_name, unk,
385  lower_bound, upper_bound);
386  }
387  u_vect.emplace_back(std::move(unk));
388  }
389  // Energies
390  this->oper_.ComputeEnergies(iter, current_time, current_time_step, u_vect);
391 
392  // Save variables for visualization
393  ProblemBase<VAR, PST>::post_processing(iter, current_time, current_time_step);
394 
395  // Save specialized values at each time-step if required
396  bool must_be_saved = this->pst_.get_enable_save_specialized_at_iter();
397  this->save_specialized(must_be_saved);
398 }
