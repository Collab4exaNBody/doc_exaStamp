170 template <class VARS, CoefficientDiscretization SCHEME, Diffusion DIFFU_NAME>
171 void DiffusionNLFormIntegrator<VARS, SCHEME, DIFFU_NAME>::AssembleElementVector(
172  const mfem::Array<const mfem::FiniteElement*>& el, mfem::ElementTransformation& Tr,
173  const mfem::Array<const mfem::Vector*>& elfun, const mfem::Array<mfem::Vector*>& elvect) {
174  int blk = 0;
175  int nd = el[blk]->GetDof();
176  int dim = el[blk]->GetDim();
177  gradPsi.SetSize(nd, dim);
178  Psi.SetSize(nd);
179  gradU.SetSize(dim);
180 
181  elvect[blk]->SetSize(nd);
182  *elvect[blk] = 0.;
183 
184  const mfem::IntegrationRule* ir =
185  &mfem::IntRules.Get(el[blk]->GetGeomType(), 2 * el[blk]->GetOrder() + Tr.OrderW());
186 
187  for (int i = 0; i < ir->GetNPoints(); i++) {
188  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
189  el[blk]->CalcShape(ip, Psi);
190  Tr.SetIntPoint(&ip);
191 
192  const auto& u = *elfun[blk] * Psi;
193 
194  // Laplacian : given u, compute (grad(u), grad(psi)), psi is shape function.
195  // given u (elfun), compute grad(u)
196  el[blk]->CalcPhysDShape(Tr, gradPsi);
197  gradPsi.MultTranspose(*elfun[blk], gradU);
198 
199  const double coeff_diffu = this->diffusion(Tr, ip, u, this->params_) * ip.weight * Tr.Weight();
200  gradU *= coeff_diffu;
201  gradPsi.AddMult(gradU, *elvect[blk]);
202  }
203 }
