780  const int& fe_order, std::tuple<Args...> tup_args,
781  std::vector<mfem::Vector> translations) {
782  a_my_class.fe_order_ = fe_order;
783  auto tup_size = std::tuple_size<decltype(tup_args)>::value;
784  a_my_class.dimension_ = 3;
785 
786  auto element = mfem::Element::TETRAHEDRON;
787  switch (Meshes::from(mesh_type)) {
788  case Meshes::InlineSquareWithTetraedres: {
789  element = mfem::Element::TETRAHEDRON;
790  break;
791  }
792  case Meshes::InlineSquareWithHexaedres: {
793  element = mfem::Element::HEXAHEDRON;
794  break;
795  }
796  default:
797  mfem::mfem_error(
798  "SpatialDiscretization::SpatialDiscretization: here, only "
799  "InlineSquareWithTetraedres, InlineSquareWithHexaedres mesh types are allowed");
800  break;
801  }
802 
803  if (tup_size == 6) {
804  a_my_class.dimension_ = 3;
805  const auto nx = std::get<0>(tup_args);
806  const auto ny = std::get<1>(tup_args);
807  const auto nz = std::get<2>(tup_args);
808  const auto sx = std::get<3>(tup_args);
809  const auto sy = std::get<4>(tup_args);
810  const auto sz = std::get<5>(tup_args);
811 
812  mfem::Mesh tmp_mesh = mfem::Mesh::MakeCartesian3D(nx, ny, nz, element, sx, sy, sz);
813 
814  const auto tol = 1.e-6;
815  std::vector<int> periodicMap = tmp_mesh.CreatePeriodicVertexMapping(translations, tol);
816  auto periodic_mesh = mfem::Mesh::MakePeriodic(tmp_mesh, periodicMap);
817  tmp_mesh.Clear();
818  mfem::Mesh tmp_mesh_periodic =
819  mfem::Mesh(periodic_mesh, true); // replace the input mesh with the periodic one
820  a_my_class.mesh_ =
821  new mfem::ParMesh(MPI_COMM_WORLD, tmp_mesh_periodic); // definition of the parallel mesh
822  tmp_mesh_periodic.Clear();
823 
824  } else {
825  std::string msg =
826  "SpatialDiscretization::SpatialDiscretization: " + mesh_type +
827  " requires six arguments, the number of nodes and the length along each direction";
828  mfem::mfem_error(msg.c_str());
829  }
830  a_my_class.mesh_max_bdr_attributes_ = a_my_class.mesh_.bdr_attributes.Max();
831  }
