139 void MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency() {
140  std::vector<mfem::ParGridFunction> aux_gf = this->get_aux_gf();
141  std::vector<std::vector<std::string>> aux_infos = this->get_aux_infos();
142 
143  //==========================================================
144  // Get chemical potentials and mobilities (aux. variables)
145  //==========================================================
146  bool temperature_found = false;
147  std::set<std::string> set_mob, set_pot, set_dpot;
148  for (std::size_t i = 0; i < aux_infos.size(); ++i) {
149  const auto& variable_info = aux_infos[i];
150 
151  MFEM_VERIFY(!variable_info.empty(), "Empty variable_info encountered.");
152  size_t vsize = variable_info.size();
153 
154  MFEM_VERIFY(vsize > 0,
155  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: at least "
156  "one additionnal information are expected for auxiliary variables associated with "
157  "this integrator");
158  const std::string& symbol = toLowerCase(variable_info.back());
159 
160  if (symbol == "mu") {
161  MFEM_VERIFY(
162  vsize == 2,
163  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: error while "
164  "getting chemical potentials. Expected [element, 'mu']");
165 
166  const std::string& elem_name = variable_info[0];
167  this->mu_gf_.emplace(toUpperCase(elem_name), std::move(aux_gf[i]));
168  this->mu_found_ = true;
169  set_pot.insert(elem_name);
170  } else if (symbol == "dmu" && this->enable_diffusion_chemical_potentials_) {
171  MFEM_VERIFY(
172  vsize == 2,
173  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: error while "
174  "getting diffusion chemical potentials. Expected [element, 'dmu']");
175 
176  const std::string& elem_name = variable_info[0];
177  this->dmu_gf_.emplace(toUpperCase(elem_name), std::move(aux_gf[i]));
178  this->dmu_found_ = true;
179  set_dpot.insert(elem_name);
180  } else if (symbol == "inter_mob") {
181  // Mobilities can be directly supplied within this integrator or overloaded by considering a
182  // child class.
183  MFEM_VERIFY(
184  vsize >= 2,
185  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: error while "
186  "getting mobilities.Expected [elem_name, 'mob'].");
187 
188  const std::string& elem_name = toUpperCase(variable_info[vsize - 2]);
189 
190  this->mob_gf_.emplace(toUpperCase(elem_name), std::move(aux_gf[i]));
191  this->mobilities_found_ = true;
192  set_mob.insert(elem_name);
193  } else if (toUpperCase(variable_info.back()) == "T") {
194  // this->temp_gf_ = std::move(aux_gf[i]);
195  // const std::string& elem_name = toUpperCase(variable_info[vsize - 2]);
196  this->temp_gf_.emplace_back(std::move(aux_gf[i]));
197  temperature_found = true;
198  }
199  }
200 
201  MFEM_VERIFY(
202  this->dmu_found_ || this->mu_found_,
203  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
204  "Neither chemical potentials, nor diffusion chemical potentials found. At least one of them "
205  "is required.");
206 
207  if (this->enable_diffusion_chemical_potentials_) {
208  MFEM_VERIFY(this->dmu_found_,
209  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
210  "Expected diffusion chemical potentials.");
211 
212  if (this->mobilities_found_) {
213  MFEM_VERIFY(this->mob_gf_.size() == this->dmu_gf_.size(),
214  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
215  "As many mobilities as diffusion chemical potentials are expected.");
216 
217  MFEM_VERIFY(set_mob == set_dpot,
218  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
219  "Elements must be the same for mobilities and diffusion chemical potentials.");
220  }
221 
222  } else {
223  MFEM_VERIFY(this->mu_found_,
224  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
225  "Expected chemical potentials.");
226  if (this->mobilities_found_) {
227  MFEM_VERIFY(this->mob_gf_.size() == this->mu_gf_.size(),
228  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
229  "As many mobilities as chemical potentials are expected.");
230 
231  MFEM_VERIFY(set_mob == set_pot,
232  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
233  "Elements must be the same for mobilities and chemical potentials.");
234  }
235  }
236  if (this->scale_coefficients_by_temperature_ || this->scale_variables_by_temperature_) {
237  MFEM_VERIFY(
238  temperature_found,
239  "MassDiffusionFluxNLFormIntegrator<VARS>::check_variables_consistency: "
240  "No temperature found. Expected temperature when scaling by temperature is required. ");
241  }

271 std::vector<double> MassDiffusionFluxNLFormIntegrator<VARS>::get_flux_coefficient(
272  const int nElement, const mfem::IntegrationPoint& ip) {
273  std::vector<double> coefficient;
274  double scaling_factor = 1.;
275  if (this->scale_coefficients_by_temperature_) {
276  scaling_factor = Physical::R * this->temp_gf_[0].GetValue(nElement, ip);
277  }
278 
279  if (this->mobilities_found_) {
280  coefficient.reserve(this->mob_gf_.size());
281  for (const auto& [component, mob_gf] : this->mob_gf_) {
282  coefficient.emplace_back(mob_gf.GetValue(nElement, ip) / scaling_factor);
283  }
284  }
285 
286  return coefficient;

348 template <class VARS>
349 void MassDiffusionFluxNLFormIntegrator<VARS>::get_cross_thermal_flux(
350  mfem::Vector& grad_pot, const mfem::ParGridFunction& potential, mfem::ElementTransformation& Tr,
351  const int nElement, const mfem::IntegrationPoint& ip, const int dim) {
352  mfem::Vector gradT;
353  gradT.SetSize(dim);
354  auto temp = SlothGridFunction(this->temp_gf_[0]);
355  const auto pot_at_ip = potential.GetValue(nElement, ip);
356  const auto temp_at_ip = this->temp_gf_[0].GetValue(nElement, ip);
357  const auto dmu_over_square_temp_at_ip = pot_at_ip / (temp_at_ip * temp_at_ip);
358 
359  temp.GetGradient(Tr, this->gradPsi, gradT);
360  gradT *= dmu_over_square_temp_at_ip;
361  grad_pot /= temp_at_ip;
362  grad_pot -= gradT;
363 }
