155  const int& it, const double& t, const double& dt, const std::vector<mfem::Vector>& u) {
156  Catch_Time_Section("DiffusionOperatorBase::ComputeEnergies");
157 
158  std::vector<mfem::ParGridFunction> vun;
159  vun.reserve(u.size());
160  for (size_t i = 0; i < u.size(); ++i) {
161  mfem::ParGridFunction un(this->fes_[i]);
162  un.SetFromTrueDofs(u[i]);
163  vun.emplace_back(std::move(un));
164  }
165 
166  std::vector<mfem::ParGridFunction*> vun_ptr;
167  for (auto& un : vun) {
168  vun_ptr.push_back(&un);
169  }
170 
171  auto gphi = this->nlfi_ptr_->get_energy(vun_ptr, 1.0);
172  auto energy_coeff = gphi.get();
173  mfem::ParGridFunction gf(this->fes_[0]);
174 
175  gf.ProjectCoefficient(*energy_coeff);
176 
177  // Calcul de l'int√©grale de l'objet FunctionCoefficient sur le domaine
178  mfem::ConstantCoefficient zero(0.);
179  const auto energy = gf.ComputeLpError(1, zero);
180  this->time_specialized_.emplace(IterationKey(it, dt, t),
181  SpecializedValue("Density[J.m-3]", energy));
182 }
